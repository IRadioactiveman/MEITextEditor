        s +=
"<h4>Syllable:</h4>\

        currentType = currentSyllable.type;

            if(currentSyllable.staff == staffs[i].n){
                staffID = staffs[i].id;
            }
        }
        s += staffID + "<br>";
        s += 
"Syllable:           <input id=\"syllabletext\" type=\"text\">"+ currentSyllable.syllable +"<br>\

        if(currentSyllable.initial){
